# Used Modules:
Add to php ini (for windows):
extension=pdo_mysql
extension=mysqli

#Enviroment Values:
HOST = sql host with port number
DB_NAME = db name
DB_USER = db username
PASS = user's password on dp
PHPRC = project folder for using php ini in this project
EXT_PATH = extenstion path (defult is {PHP install directory}\ext)
